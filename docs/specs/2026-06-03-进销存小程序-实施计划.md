**`utils/request.js` 核心逻辑：**

```javascript
function request(url, method = 'GET', data = {}) {
  return new Promise((resolve, reject) => {
    wx.request({
      url: `${BASE_URL}${url}`,
      method,
      data,
      header: { 'Authorization': `Bearer ${wx.getStorageSync('access_token')}` },
      success(res) {
        if (res.data.code === 'EXPIRED') return wx.redirectTo({ url: '/pages/lock/lock' });
        if (res.data.code === 'UNAUTHORIZED') return refreshAndRetry(url, method, data, resolve, reject);
        resolve(res.data);
      },
      fail() { wx.showToast({ title: '网络开小差了，请重试', icon: 'none' }); reject(); }
    });
  });
}
```
